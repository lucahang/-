ClientNetwork::~ClientNetwork() {
    if (m_socket != INVALID_SOCKET) closesocket(m_socket);
    if (m_udpSocket != INVALID_SOCKET) closesocket(m_udpSocket);
    WSACleanup();
}

bool ClientNetwork::ConnectToServer(const char* ip, int port) {
    // 1. СЌНг TCP (ПижЦЭЈЕР)
    m_socket = socket(AF_INET, SOCK_STREAM, 0);
    sockaddr_in addr{};
    addr.sin_family = AF_INET;
    addr.sin_port = htons(static_cast<u_short>(port));
    inet_pton(AF_INET, ip, &addr.sin_addr);

    if (connect(m_socket, reinterpret_cast<sockaddr*>(&addr), sizeof(addr)) == SOCKET_ERROR) {
        closesocket(m_socket);
        m_socket = INVALID_SOCKET;
        return false;
    }

    // 2. ЁОаТдіЁПГѕЪМЛЏ UDP Socket ВЂЗЂЫЭЮеЪжАќДђЖД
    m_udpSocket = socket(AF_INET, SOCK_DGRAM, IPPROTO_UDP);
    
    // діДѓ UDP НгЪеЛКГхЧјЃЌЗРжЙИпЧхЛ­УцЫВМфЗЂАќЙ§ЖрЕМжТЖЊАќ
    int recvBuf = 1024 * 1024 * 4; // 4MB
    setsockopt(m_udpSocket, SOL_SOCKET, SO_RCVBUF, (const char*)&recvBuf, sizeof(recvBuf));

    sockaddr_in udpAddr{};
    udpAddr.sin_family = AF_INET;
    udpAddr.sin_port = htons(port + 1); // ЗўЮёЖЫ UDP ЖЫПк
    inet_pton(AF_INET, ip, &udpAddr.sin_addr);

    // ЗЂЫЭФЇЪ§зїЮЊЮеЪжАќ
    uint32_t magic = PROTOCOL_MAGIC;
    sendto(m_udpSocket, (const char*)&magic, sizeof(magic), 0, (sockaddr*)&udpAddr, sizeof(udpAddr));
    
    std::cout << "[Client] UDP ЮеЪжАќвбЗЂЫЭ" << std::endl;
    return true;
}

// ЁОКЫаФаТдіЁПНгЪе UDP ЧаЦЌВЂЦДГЩЭъећ H.264 жЁ
bool ClientNetwork::ReceiveVideoFrameUDP(std::vector<unsigned char>& outFrame) {
    static uint32_t latestFrameId = 0;
    static std::map<uint16_t, std::vector<unsigned char>> fragments;

    char buf[2048];
    int r = recvfrom(m_udpSocket, buf, sizeof(buf), 0, NULL, NULL);
    if (r <= 0) return false;

    UdpFragmentHeader* hdr = (UdpFragmentHeader*)buf;
    if (hdr->magic != PROTOCOL_MAGIC) return false;

    // ЧщПі AЃКЪеЕНЕФЪЧОЩжЁЕФВаЦЌ -> жБНгШгЕєЃЌВЛДІРэ
    if (hdr->frameId < latestFrameId) return false;

    // ЧщПі BЃКЪеЕНЕФЪЧаТжЁЕФЗжЦЌ -> ЧхПеОЩЕФЃЌПЊЪМДцаТЕФ
    if (hdr->frameId > latestFrameId) {
        fragments.clear();
        latestFrameId = hdr->frameId;
        // етвЛВНЗЧГЃживЊЃЁВЛвЊПЈдкОЩжЁРя
    }

    // БЃДцЕБЧАЗжЦЌ
    fragments[hdr->fragIndex] = std::vector<unsigned char>(
        buf + sizeof(UdpFragmentHeader),
        buf + sizeof(UdpFragmentHeader) + hdr->payloadLen
    );

    // ЧщПі CЃКМьВщЪЧЗёМЏЦы
    if (fragments.size() == hdr->fragCount) {
        outFrame.clear();
        for (int i = 0; i < hdr->fragCount; ++i) {
            outFrame.insert(outFrame.end(), fragments[i].begin(), fragments[i].end());
        }
        fragments.clear();
        return true; // жЛгаМЏЦыСЫВХНЛИјНтТыЦї
    }

    return false; // ЛЙУЛМЏЦыЃЌМЬајЕШЯТвЛИіАќ
}

bool ClientNetwork::SendAll(const char* data, int len) {
    int sent = 0;
    while (sent < len) {
        int r = send(m_socket, data + sent, len - sent, 0);
        if (r <= 0) return false;
        sent += r;
    }
    return true;
}

void ClientNetwork::SendCommand(uint32_t eventType, int x, int y, uint32_t key, PacketType packetType) {
    if (m_socket == INVALID_SOCKET) return;

    std::lock_guard<std::mutex> lock(m_sendMutex);

    PacketHeader header{};
    header.magic = PROTOCOL_MAGIC;
    header.type = static_cast<uint32_t>(packetType);
    header.length = sizeof(ControlEvent);

    ControlEvent ev{};
    ev.eventType = eventType;
    ev.x = x;
    ev.y = y;
    ev.key = key;
    ev.type = (uint32_t)packetType;
    if (!SendAll(reinterpret_cast<const char*>(&header), sizeof(header))) return;
    SendAll(reinterpret_cast<const char*>(&ev), sizeof(ev));
}

bool ClientNetwork::ReadExactly(char* buf, int len) {
    if (len <= 0 || len > static_cast<int>(MAX_PACKET_SIZE)) return false;
    if (m_socket == INVALID_SOCKET) return false;

    int received = 0;
    while (received < len) {
        int r = recv(m_socket, buf + received, len - received, 0);
        if (r <= 0) return false;
        received += r;
    }
    return true;
}

CommandCollector::~CommandCollector() {
    running = false;

    if (hKeyHook) {
        UnhookWindowsHookEx(hKeyHook);
        hKeyHook = nullptr;
    }

    if (hookThreadId != 0) {
        PostThreadMessage(hookThreadId, WM_QUIT, 0, 0);
    }

    if (hookThread.joinable()) {
        hookThread.join();
    }

    if (s_instance == this) {
        s_instance = nullptr;
    }
}

void CommandCollector::Start(const std::string& windowTitle) {
    targetTitle = windowTitle;
    running = true;
    hookThread = std::thread(&CommandCollector::HookLoop, this);
}

void CommandCollector::HookLoop() {
    hookThreadId = GetCurrentThreadId();

    MSG dummy{};
    PeekMessage(&dummy, NULL, WM_USER, WM_USER, PM_NOREMOVE);

    while (running && hTargetWnd == NULL) {
        hTargetWnd = FindWindowA(NULL, targetTitle.c_str());
        Sleep(200);
    }

    if (!running) return;

    hKeyHook = SetWindowsHookEx(WH_KEYBOARD_LL, LowLevelKeyboardProc, GetModuleHandle(NULL), 0);
    if (!hKeyHook) {
        std::cerr << "SetWindowsHookEx failed\n";
        return;
    }

    MSG msg;
    while (running && GetMessage(&msg, NULL, 0, 0) > 0) {
        TranslateMessage(&msg);
        DispatchMessage(&msg);
    }
}

LRESULT CALLBACK CommandCollector::LowLevelKeyboardProc(int nCode, WPARAM wParam, LPARAM lParam) {
    if (nCode == HC_ACTION && s_instance) {
        if (GetForegroundWindow() == s_instance->hTargetWnd) {
            KBDLLHOOKSTRUCT* pKey = reinterpret_cast<KBDLLHOOKSTRUCT*>(lParam);

            bool isDown = (wParam == WM_KEYDOWN || wParam == WM_SYSKEYDOWN);
            bool isUp = (wParam == WM_KEYUP || wParam == WM_SYSKEYUP);

            if (isDown || isUp) {
                bool isExt = (pKey->flags & LLKHF_EXTENDED);
                s_instance->OnKeyAction(pKey->vkCode, pKey->scanCode, isDown, isExt);
            }

            return 1;
        }
    }

    return CallNextHookEx(NULL, nCode, wParam, lParam);
}

void CommandCollector::OnKeyAction(uint32_t vk, uint32_t scan, bool isDown, bool isExt) {
    if (!g_net) return;

    try {
        uint32_t eventType = isDown ? 1 : 0;
        g_net->SendCommand(eventType, 0, 0, vk, PacketType::PKT_KEYBOARD);
    }
    catch (...) {
    }
}

DisplayModule::DisplayModule() {
    m_packet = av_packet_alloc();
    m_frame = av_frame_alloc();
    m_rgbFrame = av_frame_alloc();

    const AVCodec* codec = avcodec_find_decoder(AV_CODEC_ID_H264);
    if (!codec) {
        std::cerr << "евВЛЕН H264 НтТыЦїЃЁ" << std::endl;
        return;
    }

    m_codecContext = avcodec_alloc_context3(codec);
    if (!m_codecContext) {
        std::cerr << "ЮоЗЈЗжХфНтТыЦїЩЯЯТЮФЃЁ" << std::endl;
        return;
    }

    if (avcodec_open2(m_codecContext, codec, nullptr) < 0) {
        std::cerr << "ЮоЗЈДђПЊНтТыЦїЩЯЯТЮФЃЁ" << std::endl;
        avcodec_free_context(&m_codecContext);
    }

DisplayModule::~DisplayModule() {
    if (m_codecContext) avcodec_free_context(&m_codecContext);
    if (m_packet) av_packet_free(&m_packet);
    if (m_frame) av_frame_free(&m_frame);
    if (m_rgbFrame) {
        if (m_rgbFrame->data[0]) av_freep(&m_rgbFrame->data[0]);
        av_frame_free(&m_rgbFrame);
    }
    if (m_swsContext) sws_freeContext(m_swsContext);
}

void DisplayModule::ShowFrame(const std::vector<unsigned char>& packetData) {
    if (packetData.empty() || !m_codecContext) return;

    m_packet->data = const_cast<uint8_t*>(packetData.data());
    m_packet->size = static_cast<int>(packetData.size());

    int ret = avcodec_send_packet(m_codecContext, m_packet);
    if (ret < 0) return;

    while (ret >= 0) {
        ret = avcodec_receive_frame(m_codecContext, m_frame);
        if (ret == AVERROR(EAGAIN) || ret == AVERROR_EOF) break;
        if (ret < 0) return;

        if (m_swsContext == nullptr ||
            m_currentWidth != m_frame->width ||
            m_currentHeight != m_frame->height) {

            if (m_swsContext) {
                sws_freeContext(m_swsContext);
                m_swsContext = nullptr;
            }

            if (m_rgbFrame->data[0]) {
                av_freep(&m_rgbFrame->data[0]);
            }

            m_currentWidth = m_frame->width;
            m_currentHeight = m_frame->height;

            m_swsContext = sws_getContext(
                m_currentWidth, m_currentHeight, static_cast<AVPixelFormat>(m_frame->format),
                m_currentWidth, m_currentHeight, AV_PIX_FMT_BGR24,
                SWS_FAST_BILINEAR, nullptr, nullptr, nullptr
            );

            if (!m_swsContext) return;

            av_image_alloc(
                m_rgbFrame->data,
                m_rgbFrame->linesize,
                m_currentWidth,
                m_currentHeight,
                AV_PIX_FMT_BGR24,
                1
            );
        }

        sws_scale(
            m_swsContext,
            m_frame->data,
            m_frame->linesize,
            0,
            m_currentHeight,
            m_rgbFrame->data,
            m_rgbFrame->linesize
        );

        cv::Mat img(
            m_currentHeight,
            m_currentWidth,
            CV_8UC3,
            m_rgbFrame->data[0],
            m_rgbFrame->linesize[0]
        );

        cv::imshow("Remote Control Stream", img);

        av_frame_unref(m_frame);
        av_packet_unref(m_packet);
    }
}

        // ШчЙћЖгСаТњСЫЃЌЕЏГізюОЩЕФвЛжЁЃЈЖЊжЁБЃЪЕЪБадЃЉ
        if (q.size() >= max_size) {
            q.pop();
        }
        q.push(std::move(val));
        cv.notify_one();
    }

    bool pop(T& val) {
        std::lock_guard<std::mutex> lock(m);
        if (q.empty()) return false;

        val = std::move(q.front());
        q.pop();
        return true;
    }

    bool empty() {
        std::lock_guard<std::mutex> lock(m);
        return q.empty();
    }

    // НЈвщдіМгвЛИіЧхПеЗНЗЈЃЌгУгкдкЭјТчЖЯПЊжиСЌЪБГѕЪМЛЏ
    void clear() {
        std::lock_guard<std::mutex> lock(m);
        std::queue<T>().swap(q);
    }
};

bool VideoDecoder::Init() {
    const AVCodec* codec = avcodec_find_decoder(AV_CODEC_ID_H264);
    if (!codec) return false;

    m_codecContext = avcodec_alloc_context3(codec);
    if (!m_codecContext) return false;

    if (avcodec_open2(m_codecContext, codec, NULL) < 0) return false;

    return true;
}

unsigned char* VideoDecoder::DecodeFrame(unsigned char* data, int size, int& outW, int& outH) {
    if (!m_codecContext) return nullptr;

    m_packet->data = data;
    m_packet->size = size;

    int ret = avcodec_send_packet(m_codecContext, m_packet);
    if (ret < 0) return nullptr;

    ret = avcodec_receive_frame(m_codecContext, m_frame);
    if (ret < 0) return nullptr;

    outW = m_frame->width;
    outH = m_frame->height;

    m_swsContext = sws_getCachedContext(
        m_swsContext,
        outW, outH, static_cast<AVPixelFormat>(m_frame->format),
        outW, outH, AV_PIX_FMT_BGR24,
        SWS_BICUBIC, NULL, NULL, NULL
    );

    if (!m_swsContext) return nullptr;

    if (!m_rgbFrame->data[0]) {
        av_image_alloc(m_rgbFrame->data, m_rgbFrame->linesize, outW, outH, AV_PIX_FMT_BGR24, 32);
    }

    sws_scale(m_swsContext, m_frame->data, m_frame->linesize, 0, outH,
        m_rgbFrame->data, m_rgbFrame->linesize);

    av_frame_unref(m_frame);
    av_packet_unref(m_packet);

    return m_rgbFrame->data[0];
}

void VideoDecoder::Cleanup() {
    if (m_codecContext) avcodec_free_context(&m_codecContext);
    if (m_frame) av_frame_free(&m_frame);
    if (m_rgbFrame) {
        if (m_rgbFrame->data[0]) av_freep(&m_rgbFrame->data[0]);
        av_frame_free(&m_rgbFrame);
    }
    if (m_packet) av_packet_free(&m_packet);
    if (m_swsContext) sws_freeContext(m_swsContext);
}

bool ClientNetwork::ConnectToServer(const char* ip, int port) {
    // 1. СЌНг TCP (ПижЦЭЈЕР)
    m_socket = socket(AF_INET, SOCK_STREAM, 0);
    sockaddr_in addr{};
    addr.sin_family = AF_INET;
    addr.sin_port = htons(static_cast<u_short>(port));
    inet_pton(AF_INET, ip, &addr.sin_addr);

    if (connect(m_socket, reinterpret_cast<sockaddr*>(&addr), sizeof(addr)) == SOCKET_ERROR) {
        closesocket(m_socket);
        m_socket = INVALID_SOCKET;
        return false;
    }

    // 2. ЁОаТдіЁПГѕЪМЛЏ UDP Socket ВЂЗЂЫЭЮеЪжАќДђЖД
    m_udpSocket = socket(AF_INET, SOCK_DGRAM, IPPROTO_UDP);
    
    // діДѓ UDP НгЪеЛКГхЧјЃЌЗРжЙИпЧхЛ­УцЫВМфЗЂАќЙ§ЖрЕМжТЖЊАќ
    int recvBuf = 1024 * 1024 * 4; // 4MB
    setsockopt(m_udpSocket, SOL_SOCKET, SO_RCVBUF, (const char*)&recvBuf, sizeof(recvBuf));

    sockaddr_in udpAddr{};
    udpAddr.sin_family = AF_INET;
    udpAddr.sin_port = htons(port + 1); // ЗўЮёЖЫ UDP ЖЫПк
    inet_pton(AF_INET, ip, &udpAddr.sin_addr);

    // ЗЂЫЭФЇЪ§зїЮЊЮеЪжАќ
    uint32_t magic = PROTOCOL_MAGIC;
    sendto(m_udpSocket, (const char*)&magic, sizeof(magic), 0, (sockaddr*)&udpAddr, sizeof(udpAddr));

    return true;

bool ClientNetwork::ReceiveVideoFrameUDP(std::vector<unsigned char>& outFrame) {
    static uint32_t latestFrameId = 0;
    static std::map<uint16_t, std::vector<unsigned char>> fragments;

    char buf[2048];
    int r = recvfrom(m_udpSocket, buf, sizeof(buf), 0, NULL, NULL);
    if (r <= 0) return false;

    UdpFragmentHeader* hdr = (UdpFragmentHeader*)buf;
    if (hdr->magic != PROTOCOL_MAGIC) return false;

    // ЧщПі AЃКЪеЕНЕФЪЧОЩжЁЕФВаЦЌ -> жБНгШгЕєЃЌВЛДІРэ
    if (hdr->frameId < latestFrameId) return false;

    // ЧщПі BЃКЪеЕНЕФЪЧаТжЁЕФЗжЦЌ -> ЧхПеОЩЕФЃЌПЊЪМДцаТЕФ
    if (hdr->frameId > latestFrameId) {
        fragments.clear();
        latestFrameId = hdr->frameId;
        // етвЛВНЗЧГЃживЊЃЁВЛвЊПЈдкОЩжЁРя
    }

    // БЃДцЕБЧАЗжЦЌ
    fragments[hdr->fragIndex] = std::vector<unsigned char>(
        buf + sizeof(UdpFragmentHeader),
        buf + sizeof(UdpFragmentHeader) + hdr->payloadLen
    );

    // ЧщПі CЃКМьВщЪЧЗёМЏЦы
    if (fragments.size() == hdr->fragCount) {
        outFrame.clear();
        for (int i = 0; i < hdr->fragCount; ++i) {
            outFrame.insert(outFrame.end(), fragments[i].begin(), fragments[i].end());
        }
        fragments.clear();
        return true; // жЛгаМЏЦыСЫВХНЛИјНтТыЦї
    }

    return false; // ЛЙУЛМЏЦыЃЌМЬајЕШЯТвЛИіАќ
}

bool ClientNetwork::SendAll(const char* data, int len) {
    int sent = 0;
    while (sent < len) {
        int r = send(m_socket, data + sent, len - sent, 0);
        if (r <= 0) return false;
        sent += r;
    }
    return true;
}

void ClientNetwork::SendCommand(uint32_t eventType, int x, int y, uint32_t key, PacketType packetType) {
    if (m_socket == INVALID_SOCKET) return;

    std::lock_guard<std::mutex> lock(m_sendMutex);

    PacketHeader header{};
    header.magic = PROTOCOL_MAGIC;
    header.type = static_cast<uint32_t>(packetType);
    header.length = sizeof(ControlEvent);

    ControlEvent ev{};
    ev.eventType = eventType;
    ev.x = x;
    ev.y = y;
    ev.key = key;
    ev.type = (uint32_t)packetType;
    if (!SendAll(reinterpret_cast<const char*>(&header), sizeof(header))) return;
    SendAll(reinterpret_cast<const char*>(&ev), sizeof(ev));
}

bool ClientNetwork::ReadExactly(char* buf, int len) {
    if (len <= 0 || len > static_cast<int>(MAX_PACKET_SIZE)) return false;
    if (m_socket == INVALID_SOCKET) return false;

    int received = 0;
    while (received < len) {
        int r = recv(m_socket, buf + received, len - received, 0);
        if (r <= 0) return false;
        received += r;
    }
    return true;
}

    targetTitle = windowTitle;
    running = true;
    hookThread = std::thread(&CommandCollector::HookLoop, this);
}

void CommandCollector::HookLoop() {
    hookThreadId = GetCurrentThreadId();

    MSG dummy{};
    PeekMessage(&dummy, NULL, WM_USER, WM_USER, PM_NOREMOVE);

    while (running && hTargetWnd == NULL) {
        hTargetWnd = FindWindowA(NULL, targetTitle.c_str());
        Sleep(200);
    }

    if (!running) return;

    hKeyHook = SetWindowsHookEx(WH_KEYBOARD_LL, LowLevelKeyboardProc, GetModuleHandle(NULL), 0);
    if (!hKeyHook) {
        std::cerr << "SetWindowsHookEx failed\n";
        return;
    }

    MSG msg;
    while (running && GetMessage(&msg, NULL, 0, 0) > 0) {
        TranslateMessage(&msg);
        DispatchMessage(&msg);
    }
}

LRESULT CALLBACK CommandCollector::LowLevelKeyboardProc(int nCode, WPARAM wParam, LPARAM lParam) {
    if (nCode == HC_ACTION && s_instance) {
        if (GetForegroundWindow() == s_instance->hTargetWnd) {
            KBDLLHOOKSTRUCT* pKey = reinterpret_cast<KBDLLHOOKSTRUCT*>(lParam);

            bool isDown = (wParam == WM_KEYDOWN || wParam == WM_SYSKEYDOWN);
            bool isUp = (wParam == WM_KEYUP || wParam == WM_SYSKEYUP);

            if (isDown || isUp) {
                bool isExt = (pKey->flags & LLKHF_EXTENDED);
                s_instance->OnKeyAction(pKey->vkCode, pKey->scanCode, isDown, isExt);
            }

            return 1;
        }
    }

    return CallNextHookEx(NULL, nCode, wParam, lParam);
}

                g_running = false;
                break;
            }

            if (videoQueue.empty()) {
                std::this_thread::sleep_for(std::chrono::milliseconds(1));
            }
        }

        std::cout << "[Client] СЌНгвбЖЯПЊЃЌЧхРэзЪдД..." << std::endl;

        // ========== ЯпГЬАВШЋЭЫГі ==========
        g_running = false;

        if (netThread.joinable()) {
            netThread.join();
        }

        // ========== ЧхПеВаСєЪгЦЕжЁ ==========
        {
            std::vector<unsigned char> tmp;
            while (videoQueue.pop(tmp));
        }

        // ========== ЪЭЗХЭјТчЖдЯѓ ==========
        g_net.reset();

        std::cout << "[Client] " << retryDelay << "УыКѓжиСЌ..." << std::endl;
        std::this_thread::sleep_for(std::chrono::seconds(retryDelay));

        // жИЪ§ЭЫБм
        retryDelay = std::min(retryDelay * 2, 30);
    }

    cv::destroyAllWindows();
    return 0;
}

#pragma once
#include <queue>

#include <utility>

template<typename T>
class SafeQueue {
private:
    std::queue<T> q;
    std::mutex m;
    // дЖПиНЈвщЩшЮЊ 1 Лђ 2ЃЌБЃжЄМЋжТЕФЪЕЪБад
    const size_t max_size = 2;

public:
    // БиаыМгЫјЃЌЗёдђЖрЯпГЬЖСШЁ size ЛсБРРЃ
    int size() {
        std::lock_guard<std::mutex> lock(m);
        return (int)q.size();
    }

    void push(T val) {
        std::lock_guard<std::mutex> lock(m);
        // ШчЙћЖгСаТњСЫЃЌЕЏГізюОЩЕФвЛжЁЃЈЖЊжЁБЃЪЕЪБадЃЉ
        if (q.size() >= max_size) {
            q.pop();
        }
        q.push(std::move(val));
        cv.notify_one();
    }

    bool pop(T& val) {
        std::lock_guard<std::mutex> lock(m);
        if (q.empty()) return false;

        val = std::move(q.front());
        q.pop();
        return true;
    }

    bool empty() {
        std::lock_guard<std::mutex> lock(m);
        return q.empty();
    }

    // НЈвщдіМгвЛИіЧхПеЗНЗЈЃЌгУгкдкЭјТчЖЯПЊжиСЌЪБГѕЪМЛЏ
    void clear() {
        std::lock_guard<std::mutex> lock(m);
        std::queue<T>().swap(q);
    }
};

VideoDecoder::VideoDecoder() {
    m_packet = av_packet_alloc();
    m_frame = av_frame_alloc();
    m_rgbFrame = av_frame_alloc();
}

VideoDecoder::~VideoDecoder() {
    Cleanup();
}

bool VideoDecoder::Init() {
    const AVCodec* codec = avcodec_find_decoder(AV_CODEC_ID_H264);
    if (!codec) return false;

    m_codecContext = avcodec_alloc_context3(codec);
    if (!m_codecContext) return false;

    if (avcodec_open2(m_codecContext, codec, NULL) < 0) return false;

    return true;
}

unsigned char* VideoDecoder::DecodeFrame(unsigned char* data, int size, int& outW, int& outH) {
    if (!m_codecContext) return nullptr;

    m_packet->data = data;
    m_packet->size = size;

    int ret = avcodec_send_packet(m_codecContext, m_packet);
    if (ret < 0) return nullptr;

    ret = avcodec_receive_frame(m_codecContext, m_frame);
    if (ret < 0) return nullptr;

    outW = m_frame->width;
    outH = m_frame->height;

    m_swsContext = sws_getCachedContext(
        m_swsContext,
        outW, outH, static_cast<AVPixelFormat>(m_frame->format),
        outW, outH, AV_PIX_FMT_BGR24,
        SWS_BICUBIC, NULL, NULL, NULL
    );

    if (!m_swsContext) return nullptr;

    if (!m_rgbFrame->data[0]) {
        av_image_alloc(m_rgbFrame->data, m_rgbFrame->linesize, outW, outH, AV_PIX_FMT_BGR24, 32);
    }

    sws_scale(m_swsContext, m_frame->data, m_frame->linesize, 0, outH,
        m_rgbFrame->data, m_rgbFrame->linesize);

    av_frame_unref(m_frame);
    av_packet_unref(m_packet);

    return m_rgbFrame->data[0];
}

void VideoDecoder::Cleanup() {
    if (m_codecContext) avcodec_free_context(&m_codecContext);
    if (m_frame) av_frame_free(&m_frame);
    if (m_rgbFrame) {
        if (m_rgbFrame->data[0]) av_freep(&m_rgbFrame->data[0]);
        av_frame_free(&m_rgbFrame);
    }
    if (m_packet) av_packet_free(&m_packet);
    if (m_swsContext) sws_freeContext(m_swsContext);
}
